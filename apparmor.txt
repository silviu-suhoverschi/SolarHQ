#include <tunables/global>

profile solarhq flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Allow network access (required for HA Supervisor API and Ingress)
  network inet tcp,
  network inet udp,
  network inet6 tcp,
  network inet6 udp,

  # Allow file capabilities
  capability,

  # Process execution
  /usr/bin/python3* ix,
  /usr/bin/node* ix,
  /usr/local/bin/python3* ix,
  /usr/local/bin/uvicorn ix,
  /command/** ix,
  /usr/bin/** ix,
  /usr/local/bin/** ix,

  # s6-overlay
  /etc/s6-overlay/** r,
  /run/s6/** rw,
  /run/service/** rw,
  /var/run/** rw,

  # Application data
  /app/** r,
  /app/static/** r,

  # SQLite database in /data
  /data/** rw,
  /data/solarhq.db-wal rw,
  /data/solarhq.db-shm rw,

  # Environment and config
  /etc/solarhq.env rw,
  /proc/*/net/if_inet6 r,
  /proc/sys/net/** r,

  # Deny sensitive paths
  deny /etc/shadow r,
  deny /etc/passwd w,
  deny /root/** rw,
}
