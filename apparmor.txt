#include <tunables/global>

profile solarhq flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Broad file access (read/write/append/lock, NOT execute)
  file,

  # Signal handling (for s6-overlay service control)
  signal (send) set=(kill,term,int,hup,cont),

  # s6-overlay entry point and runtime
  /init                  ix,
  /bin/**                ix,
  /usr/bin/**            ix,
  /usr/local/bin/**      ix,
  /command/**            ix,
  /package/**            ix,
  /run/s6/**             rwix,

  # Deny sensitive paths
  deny /etc/shadow r,
  deny /etc/passwd w,
  deny /root/** rw,
  deny /proc/*/net/if_inet6 r,
  deny /proc/*/net/ipv6_route r,
}
